character_event = {
	id = triforce.3
	desc = "DESC_TF3"
	picture = "GFX_evt_comet"
	
	trigger = {
		OR = {
			AND = {
				trait = triforce_power
				OR = {
					trait = feeble
					trait = infirm
					trait = incapable
					trait = coward
				}
			}
			AND = {
				trait = triforce_courage
				OR = {
					trait = greedy
					trait = envious
					trait = infirm
					trait = incapable
					trait = coward
					trait = cruel
					trait = arbitrary
				}
			}
			AND = {
				trait = triforce_wisdom
				OR = {
					trait = wroth
					trait = gluttonous
					trait = infirm
					trait = incapable
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		if = {
			limit = {
				trait = triforce_power
			}
		remove_trait = triforce_power
		}
		if = {
			limit = {
				trait = triforce_courage
			}
		remove_trait = triforce_courage
		}
		if = {
			limit = {
				trait = triforce_wisdom
			}
		remove_trait = triforce_wisdom
		}
	}
	option = { name = "OPTF3A" } #what have I done to deserve this!
}
