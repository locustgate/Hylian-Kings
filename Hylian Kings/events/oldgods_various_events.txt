	desc = EVTDESC_TOG_3000

			culture = geozard

		NOT = { trait = berserker }
		OR = {

			trait = aggressive_leader
		}

	weight_multiplier = {
		days = 1

		}

			is_strong_trigger = yes
			factor = 1.2
		}
		modifier = {
			factor = 0.5
			trait = craven
		}
