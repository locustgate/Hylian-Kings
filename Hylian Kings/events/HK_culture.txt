character_event = {
	id = thieving.1
	desc = "thieving1_DESC"
	picture = "GFX_evt_church"

	is_triggered_only = yes

	option = {
		name = "thieving1A"
		trigger = {
			OR = {
				war = no
				NOT = { demesne_size = 1 }
			}	
			OR = {
				liege = { war = no }
				demesne_size = 1
			}	
		}
		add_trait = on_thieving
		hidden_tooltip = {
			set_character_flag = crime_spree_begins
			character_event = { id = thieving.2 days = 120 }
		}
		any_opinion_modifier_target = {
			limit = {
				trait = on_thieving
			}
			set_character_flag = crime_spree_begins
			add_trait = on_thieving
			tooltip = { }
		}	
	}
	option = {
		name = "thieving1B"
		trigger = {
			OR = {
				AND = {
					war = yes
					demesne_size = 1 
				}	
				AND = {
					liege = { war = yes }
					NOT = { demesne_size = 1 }
				}	
			}
		}
		prestige = -20
		hidden_tooltip = {
			remove_trait = on_thieving
		}
		clr_character_flag = do_not_disturb
	}
}
